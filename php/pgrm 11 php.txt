<?php
// Prompt the user to enter a number between 1 and 7
echo "Enter a number (1-7): ";
$dayNumber = trim(fgets(STDIN));

// Determine the day of the week using a switch statement
switch ($dayNumber) {
    case 1:
        echo "Monday";
        break;
    case 2:
        echo "Tuesday";
        break;
    case 3:
        echo "Wednesday";
        break;
    case 4:
        echo "Thursday";
        break;
    case 5:
        echo "Friday";
        break;
    case 6:
        echo "Saturday";
        break;
    case 7:
        echo "Sunday";
        break;
    default:
        echo "Invalid number! Please enter a number between 1 and 7.";
}
?>

output:
Enter a number (1-7): 6
Saturday
