<?php
// Function to find and return the highest and lowest values in an array
function findMinMax($numbers) {
    // Ensure the array is not empty
    if (empty($numbers)) {
        return ['error' => 'The array is empty.'];
    }

    // Use PHP's built-in min() and max() functions
    $minValue = min($numbers);
    $maxValue = max($numbers);

    return ['min' => $minValue, 'max' => $maxValue];
}

// Sample array of numbers
$numbers = [12, 45, 7, 89, 23, 56];

// Call the function
$result = findMinMax($numbers);

// Display the results
if (isset($result['error'])) {
    echo $result['error'];
} else {
    echo "The lowest value is: " . $result['min'] . "<br>";
    echo "The highest value is: " . $result['max'];
}
?>

output:
The lowest value is: 7<br>The highest value is: 89
